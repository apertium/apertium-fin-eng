    Macros

    Macros are helper functions in apertium transfer files.

    compare

    Parametres: 1
    1. let $comparison ≔ <pos>
      1. if
        1. sl[1]['a_comp'] ≟ <comp>
        3. then:
        1. let tl[1]['a_comp'] ≔ <comp>
        2. let $comparison ≔ <com>
      3. elseif
        1. sl[1]['a_comp'] ≟ <sup>
        5. then:
        1. let tl[1]['a_comp'] ≔ <sup>
        2. let $comparison ≔ <sup>
      5. else:
        1. let $comparison ≔ <pos>

    possmangler1

    Parametres: 1
    1. let $possession ≔ <pxsp3>
      1. if
        1. sl[1]['a_num'] ≟ <sg>
        2. sl[1]['a_pers'] ≟ <p1>
        3. then:
        1. let $possession ≔ <pxsg1>
      3. elseif
        1. sl[1]['a_num'] ≟ <sg>
        2. sl[1]['a_pers'] ≟ <p2>
        5. then:
        1. let $possession ≔ <pxsg2>
      5. elseif
        1. sl[1]['a_pers'] ≟ <p3>
        7. then:
        1. let $possession ≔ <pxsp3>
      7. elseif
        1. sl[1]['a_num'] ≟ <pl>
        2. sl[1]['a_pers'] ≟ <p2>
        9. then:
        1. let $possession ≔ <pxpl2>
      9. elseif
        1. sl[1]['a_num'] ≟ <pl>
        2. sl[1]['a_pers'] ≟ <p1>
        11. then:
        1. let $possession ≔ <pxpl1>
      11. elseif
        1. sl[1]['lem'] ≟ "my"
        13. then:
        1. let $possession ≔ <pxsg1>
      13. elseif
        1. sl[1]['lem'] ≟ "your"
        15. then:
        1. let $possession ≔ <pxsg2>
      15. elseif
        1. sl[1]['lem'] ≟ "his"
        17. then:
        1. let $possession ≔ <pxsp3>
      17. elseif
        1. sl[1]['lem'] ≟ "her"
        19. then:
        1. let $possession ≔ <pxsp3>
      19. elseif
        1. sl[1]['lem'] ≟ "our"
        21. then:
        1. let $possession ≔ <pxpl1>
      21. elseif
        1. sl[1]['lem'] ≟ "their"
        23. then:
        1. let $possession ≔ <pxsp3>
      23. else:
        1. let $possession ≔ <pxsp3>

    test

    Parametres: 1
    1. let $number ≔ ""

    tensemangle1

    Parametres: 1
    1. let $conneg ≔ <conneg>
      1. if
        1. sl[1]['a_tens'] ≟ <pres>
        3. then:
        1. let $tense ≔ <pri>
      3. elseif
        1. sl[1]['a_tens'] ≟ <cni>
        5. then:
        1. let $tense ≔ <pri>
      5. elseif
        1. sl[1]['a_tens'] ≟ <past>
        7. then:
        1. let $tense ≔ <past>
        2. let $conneg ≔ <sg.conneg>
      7. else:
        1. let $tense ≔ <pri>

    personmangle1

    Parametres: 1
      1. if
        1. sl[1]['a_pers'] ≟ ""
        3. then:
        1. let $pers ≔ sl[1]['a_pers']
      3. else:
        1. let $pers ≔ <p3>
      1. if
        1. sl[1]['a_num'] ≟ ""
        3. then:
        1. let $persnum ≔ sl[1]['a_num']
      3. else:
        1. let $persnum ≔ <sg>

    artmangle1

    Parametres: 1
      1. if
        1. sl[1]['lem'] ≟ "the"
        2. sl[1]['lem'] ≟ "a"
        3. sl[1]['lem'] ≟ "The"
        4. sl[1]['lem'] ≟ "A"
        3. then:
        1. let $determiner ≔ ""
      3. else:
        1. let $determiner ≔ tl[1]['whole'] <2>

    numbermangle1

    Parametres: 1
      1. if
        1. sl[1]['a_num'] ≟ ""
        3. then:
        1. let $number ≔ sl[1]['a_num']
      3. else:
        1. let $number ≔ <sg>
