    Macros

    Macros are helper functions in apertium transfer files.

    proc-synlabel1

    Parametres: 1
    1. let tl[1]['synfunc'] ≔ ""

    person-handler1

    Parametres: 1
      1. if
        1. tl[1]['a_verb'] ≟ <vaux.actv>
        2. tl[1]['a_nbr'] ≟ <sg>
        3. tl[1]['a_person'] ≟ <p3>
        3. then:
        1. let tl[1]['a_nbr'] ≔ ""
        2. let tl[1]['a_person'] ≔ ""
    2. let tl[1]['a_voice'] ≔ ""

    mood-handler1

    Parametres: 1
    1. let tl[1]['synfunc'] ≔ ""
      1. if
        1. sl[1]['moodtense'] ≟ <pri>
        2. sl[1]['moodtense'] ≟ <cond>
        3. sl[1]['moodtense'] ≟ <imp>
        4. sl[1]['moodtense'] ≟ <pot>
        3. then:
        1. let $screeve ≔ <pres>
      3. elseif
        1. sl[1]['moodtense'] ≟ <past>
        5. then:
        1. let $screeve ≔ <past>
      5. else:
        1. let $screeve ≔ <inf>

    case-handler1

    Parametres: 1
    1. let tl[1]['synfunc'] ≔ ""
      1. if
        1. tl[1]['a_prn'] ≟ <prn.pers>
        2. sl[1]['a_cas'] ≟ <ine>
        3. sl[1]['a_cas'] ≟ <ela>
        4. sl[1]['a_cas'] ≟ <ill>
        5. sl[1]['a_cas'] ≟ <par>
        6. sl[1]['a_cas'] ≟ <ade>
        7. sl[1]['a_cas'] ≟ <abl>
        8. sl[1]['a_cas'] ≟ <all>
        9. sl[1]['a_cas'] ≟ <abe>
        10. sl[1]['a_cas'] ≟ <ins>
        11. sl[1]['a_cas'] ≟ <com>
        12. sl[1]['a_cas'] ≟ <ess>
        3. then:
        1. let tl[1]['a_cas'] ≔ <acc>
        2. let $sntype ≔ <SN>
      3. elseif
        1. sl[1]['a_cas'] ≟ <nom>
        2. sl[1]['a_cas'] ≟ <ine>
        3. sl[1]['a_cas'] ≟ <ela>
        4. sl[1]['a_cas'] ≟ <ill>
        5. sl[1]['a_cas'] ≟ <par>
        6. sl[1]['a_cas'] ≟ <ade>
        7. sl[1]['a_cas'] ≟ <abl>
        8. sl[1]['a_cas'] ≟ <all>
        9. sl[1]['a_cas'] ≟ <abe>
        10. sl[1]['a_cas'] ≟ <ins>
        11. sl[1]['a_cas'] ≟ <com>
        12. sl[1]['a_cas'] ≟ <ess>
        5. then:
        1. let tl[1]['a_cas'] ≔ ""
        2. let $sntype ≔ <SN>
      5. elseif
        1. tl[1]['a_cas'] ≟ <gen>
        2. tl[1]['a_prn'] ≟ <prn.pers>
        7. then:
        1. let tl[1]['a_cas'] ≔ ""
        2. let $sntype ≔ <SN.GEN>
      1. if
        1. sl[1]['a_cas'] ≟ <nom>
        3. then:
        1. let tl[1]['a_prnpers'] ≔ <subj>
      3. else:
        1. let tl[1]['a_prnpers'] ≔ <obj>

    1. quant
    2. nominal

    Action:

    1. case-handler1( $1 )
